cels2fahr-1.rb

def cels2fahr(cels)
  fahr = cels * 9 / 5 + 32
# print("Kashi:",fahr,"\n")
  print("Sesshi:",cels,"\t","Kashi:",fahr,"\n")
end

cels2fahr-2.rb
def cels2fahr(cels)
  fahr = cels * 9 / 5 + 32
  print("Sesshi:",cels,"\t","Kashi:",fahr,"\n")
end

cels = ARGV[0]
p cels
cels2fahr(cels.to_f)
